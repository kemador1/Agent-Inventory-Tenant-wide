# Expresiones recomendadas - Power Automate

[Compose RecipientMode]
toUpper(
  if(
    not(equals(item()?['kyn_reportmodecode'], null)),
    if(equals(int(item()?['kyn_reportmodecode']), 148250001), 'CREATED_ONLY',
      if(equals(int(item()?['kyn_reportmodecode']), 148250002), 'CREATED_NON_SYSTEM',
        if(equals(int(item()?['kyn_reportmodecode']), 148250003), 'CHANGES_ONLY',
          if(equals(int(item()?['kyn_reportmodecode']), 148250004), 'OFF', 'ALL')
        )
      )
    ),
    if(
      empty(string(item()?['kyn_reportmode'])),
      'ALL',
      trim(string(item()?['kyn_reportmode']))
    )
  )
)

[Condition Send If Recipient Valid]
@and(
  contains(trim(string(item()?['kyn_recipientemail'])), '@'),
  not(equals(outputs('Compose_RecipientMode'), 'NONE')),
  not(equals(outputs('Compose_RecipientMode'), 'OFF'))
)

[Notas]
- No usar empty() sobre integer (Choice Dataverse).
- Para Choice: validar null con equals(..., null).
- Mantener fallback en kyn_reportmode mientras existan datos legacy.
